function addEltToSVG(svg, name, attrs) { var element = document.createElementNS("http://www.w3.org/2000/svg", name); if (attrs === undefined) attrs = {}; for (var key in attrs) { element.setAttributeNS(null, key, attrs[key]); } svg.appendChild(element); }
